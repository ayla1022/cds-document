#### Node.js原生示例代码

shell 安装驱动包：

```
npm install mongodb --save
( 若安装不成功可以尝试更换源，npm config set registry http://registry.cnpmjs.org )
npm init
```

示例代码：

```
'use strict';

var mongoClient = require('mongodb').MongoClient,
    assert = require('assert');

// 拼接 URI
var url = 'mongodb://username:password@10.xx.xx.xx:27017/admin';

mongoClient.connect(url, function(err, db) {
    assert.equal(null, err);
    var db = db.db('testdb'); // 选择一个 db
    var col = db.collection('demoCol'); // 选择一个集合(表)
   // 插入数据
    col.insertOne(
        {
            a: 1,
            something: "yy"
        },
        //可选参数
        //{
        //    w: 'majority' // 开启 “大多数”模式，保证数据写入 Secondary 节点
        //},
        function(err, r) {
            console.info("err:", err);
            assert.equal(null, err);
            // 断言写入成功
            assert.equal(1, r.insertedCount);
            // 查询数据
            col.find().toArray(function(err, docs) {
                assert.equal(null, err);
                console.info("docs:", docs);
                db.close();
            });
        }
    );
});
```

输出：

```
[root@VM_2_167_centos node]# node index.js
docs: [ { _id: 567a1bf26773935b3ff0b42a, a: 1, something: 'yy' } ]
```

```
var dbUri = "mongodb://" + username + ":" + password + "@" + host + ":" + port + "/" + dbName;
var opts = {
    auth: {
        authSource: 'admin'
    }
};
var connection = mongoose.createConnection(dbUri, opts);
```
